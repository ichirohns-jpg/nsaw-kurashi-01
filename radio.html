// ========================================== // NSAW暮らし情報局 共通データ配信・制御コアロジック // 対象: kawaraban.html / radio.html / app.html // ========================================== const SHEET_ID = '149lAFx3oPGjB0UqDpPWZJqt_bBeDWMPd-Rfc6LXQgoc'; const SHEET_NAME = '4市瓦版'; // ※各ページの対象シート名に書き換えて共通利用 const url = 'https://docs.google.com/spreadsheets/d/' + SHEET_ID + '/gviz/tq?sheet=' + encodeURIComponent(SHEET_NAME) + '&tqx=out:json'; function cell(r, i) { return r.c[i] ? (r.c[i].v || r.c[i].f || '') : ''; } // カードHTML生成共通関数 function makeCard(item) { return `
${item.date}
${item.title}
${item.text}
`; } fetch(url) .then(res => res.text()) .then(text => { const json = JSON.parse(text.substring(47).slice(0, -2)); const rows = json.table.rows; let items = []; rows.forEach((r) => { // 主要データがすべて空の不正行・空行を完全に除外 if (!cell(r, 0) && !cell(r, 2) && !cell(r, 3)) return; items.push({ date: cell(r, 0), image: cell(r, 1), title: cell(r, 2), text: cell(r, 3) }); }); // 【② データ順統一】reverse禁止。指定のミリ秒変換による安全日付降順ソート items.sort((a, b) => { const aTime = Number(new Date(a.date || '')); const bTime = Number(new Date(b.date || '')); return (isNaN(bTime) ? 0 : bTime) - (isNaN(aTime) ? 0 : aTime); }); // 【① URLルール統一】パラメータはすべて「id」に完全統一 const idParam = new URLSearchParams(location.search).get("id"); let target = null; if (idParam) { // 配列インデックス（-1）アクセスに統一（find / originalIndexは禁止） const idx = parseInt(idParam, 10) - 1; // 範囲外チェック if (idx >= 0 && idx < items.length) { target = items[idx]; } } // パラメータ指定はあるが、該当データが存在しない（範囲外）場合 if (idParam && !target) { document.getElementById('latestArea').innerHTML = `
指定された記事が見つかりません
`; return; } // 表示ルール統一（targetがあれば1件表示、なければ全件表示） document.getElementById('latestArea').innerHTML = target ? makeCard(target) : items.map(i => makeCard(i)).join(''); }) .catch(() => { document.getElementById('latestArea').innerHTML = `
読み込みに失敗しました
`; });
